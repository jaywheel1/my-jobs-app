        function filterJobs() {

            var categoryFilter = document.getElementById('categoryFilter').value.toLowerCase();

            var locationFilter = document.getElementById('locationFilter').value.toLowerCase();

            var jobTypeFilter = document.getElementById('jobTypeFilter').value.toLowerCase();

            var searchInput = document.getElementById('searchInput').value.toLowerCase();


            var filteredJobs = allJobs.filter(function(job) {

                var matchesCategory = !categoryFilter || job['Category'].toLowerCase().indexOf(categoryFilter) !== -1;

                var matchesLocation = !locationFilter || job['Location'].toLowerCase().indexOf(locationFilter) !== -1;

                var matchesJobType = !jobTypeFilter || job['Job Type'].toLowerCase().indexOf(jobTypeFilter) !== -1;

                var matchesSearch = !searchInput || 

                    job['Job Title'].toLowerCase().indexOf(searchInput) !== -1 ||

                    job['Company'].toLowerCase().indexOf(searchInput) !== -1 ||

                    job['Description'].toLowerCase().indexOf(searchInput) !== -1;


                return matchesCategory && matchesLocation && matchesJobType && matchesSearch;

            });


            displayJobs(filteredJobs);

        }


        function startAutoRefresh() {

                }

            }, AUTO_REFRESH_INTERVAL);

        }

        function openApplication(jobTitle, company) {

            if (!jobTitle || !company) {

                console.error('Invalid job data:', { jobTitle: jobTitle, company: company });

                alert('Error: Invalid job information. Please try again.');

                return;

            }

            

            currentJob = { title: jobTitle, company: company };

            document.getElementById('applicationModal').style.display = 'block';

            document.getElementById('applicationForm').reset();

        }


        function closeModal() {

            document.getElementById('applicationModal').style.display = 'none';

            document.getElementById('applicationForm').reset();

        }


        function handleFormSubmission(e) {

            e.preventDefault();

            

            var submitBtn = document.getElementById('submitBtn');

            var originalText = submitBtn.innerHTML;

            

            submitBtn.disabled = true;

            submitBtn.innerHTML = '⏳ Submitting...';


            try {

                var formData = new FormData(e.target);

                var applicationData = {

                    'Job Title': currentJob ? currentJob.title : 'Unknown',

                    'Company': currentJob ? currentJob.company : 'Unknown',

                    'Applicant Name': formData.get('applicantName') || '',

                    'Email': formData.get('applicantEmail') || '',

                    'Telegram Handle': formData.get('applicantTelegram') || '',

                    'Location': formData.get('applicantLocation') || '',

                    'CV/Resume Link': formData.get('applicantCV') || '',

                    'Date Applied': new Date().toISOString().split('T')[0]

                };


                var requiredFields = ['Applicant Name', 'Email', 'Telegram Handle', 'Location', 'CV/Resume Link'];

                var missingFields = requiredFields.filter(function(field) {

                    return !applicationData[field];

                });

                

                if (missingFields.length > 0) {

                    throw new Error('Please fill in all required fields: ' + missingFields.join(', '));

                }


                submitApplication(applicationData).then(function(result) {

                    if (result.success) {

                        alert('🎉 Application submitted successfully! We\'ll be in touch soon.');

                        closeModal();

                    } else {

                        throw new Error(result.message || 'Submission failed');

                    }

                }).catch(function(error) {

                    console.error('Error submitting application:', error);

                    alert('❌ ' + (error.message || 'There was an error submitting your application. Please try again.'));

                }).finally(function() {

                    submitBtn.disabled = false;

                    submitBtn.innerHTML = originalText;

                });

                

            } catch (error) {

                console.error('Error submitting application:', error);

                alert('❌ ' + (error.message || 'There was an error submitting your application. Please try again.'));

                submitBtn.disabled = false;

                submitBtn.innerHTML = originalText;

            }

        }


        function submitApplication(data) {

            return new Promise(function(resolve, reject) {

                try {

                    console.log('Application data:', data);

                    

                    setTimeout(function() {

                        resolve({ success: true });

                    }, 1000);

                    

                } catch (error) {

                    console.error('Submission error:', error);

                    reject(error);

                }

            });

        }
